// FEATURE DEMONSTRATED: Conditional Statements
// KEY CONCEPTS: If statements, conditional execution, variable modification

class Vehicle {
    vehicleId : int = 1001;
    fuelLevel : int = 75;
    needsRefuel : bool = False;
    
    function checkAndRefuel(amount : int) : int {
        // Add fuel to current level
        fuelLevel = fuelLevel + amount;
        
        // Check if fuel is above 90%
        if (fuelLevel > 90) {
            needsRefuel = False;
        }
        
        // Check if fuel is at or below 90%
        if (fuelLevel <= 90) {
            needsRefuel = True;
        }
        
        return fuelLevel;
    }
    
    // Conditional logic demonstration
    fuel1 : int = checkAndRefuel(10);  // 75 + 10 = 85, needsRefuel = True
    fuel2 : int = checkAndRefuel(20);  // 85 + 20 = 105, needsRefuel = False
}

// FINAL STATE: vehicleId=1001, fuelLevel=105, needsRefuel=False, fuel1=85, fuel2=105
