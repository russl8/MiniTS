// FEATURE DEMONSTRATED: Comparison Operators
// KEY CONCEPTS: Comparison operators (==, !=, <, >, <=, >=), conditional logic

class BankAccount {
    balance : int = 500;
    minBalance : int = 100;
    
    function checkStatus(dummy : int) : char {
        result : char = 'L';
        
        // Check if balance is above minimum
        if (balance > minBalance) {
            result = 'A';
        }
        
        // Check if balance equals minimum
        if (balance == minBalance) {
            result = 'W';
        }
        
        return result;
    }
    
    function compareBalance(other : int) : bool {
        // Multiple comparison operators demonstration
        isHigher : bool = balance > other;
        isEqual : bool = balance == other;
        isLower : bool = balance < other;
        
        return isHigher;
    }
    
    // Comparison operations
    status : char = checkStatus(1);           // Returns 'A' (Active)
    comparison : bool = compareBalance(400);  // Returns True (500 > 400)
}

// FINAL STATE: balance=500, minBalance=100, status='A', comparison=True
