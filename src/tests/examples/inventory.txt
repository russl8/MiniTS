// FEATURE DEMONSTRATED: Class Inheritance
// KEY CONCEPTS: Parent-child classes, extends keyword, inheritance

class Product {
    productId : int = 101;
    quantity : int = 50;
    
    function updateQuantity(newQty : int) : int {
        // Update product quantity
        quantity = newQty;
        return quantity;
    }
}

class Warehouse extends Product {
    location : char = 'A';
    
    function getLocation(dummy : int) : char {
        // Access warehouse-specific property
        return location;
    }
    
    // Child class can call parent methods and access parent variables
    newQuantity : int = updateQuantity(75);    // Inherited method
    currentLocation : char = getLocation(1);   // Child method
}

// FINAL STATE: productId=101, quantity=75, location='A', newQuantity=75, currentLocation='A'
