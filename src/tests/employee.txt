// FEATURE DEMONSTRATED: Boolean Operations
// KEY CONCEPTS: Boolean logic, true/false values, logical operators

class Employee {
    isActive : bool = True;
    isVerified : bool = False;
    accessLevel : int = 5;
    
    function checkAccess(requiredLevel : int) : bool {
        // Boolean AND operation with comparison
        hasAccess : bool = isActive && accessLevel >= requiredLevel;
        return hasAccess;
    }
    
    function updateStatus(newActive : bool) : bool {
        // Update employee status
        isActive = newActive;
        return isActive;
    }
    
    // Boolean operations
    accessGranted : bool = checkAccess(3);     // True && (5 >= 3) = True
    currentStatus : bool = updateStatus(False); // Sets isActive to False
}

// FINAL STATE: isActive=False, isVerified=False, accessLevel=5, accessGranted=True, currentStatus=False
