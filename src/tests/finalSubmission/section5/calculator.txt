// FEATURE DEMONSTRATED: Conditional Statements
// KEY CONCEPTS: If statements, conditional execution, branching logic

class Calculator {
    value : int = 25;
    threshold : int = 20;
    
    function evaluate(testValue : int) : char {
        result : char = 'L';
        
        // Check for high category
        if (testValue >= 30) {
            result = 'H';
        }
        
        // Check for medium category
        if (testValue >= 20) {
            result = 'M';
        }
        
        return result;
    }
    
    function compare(input : int) : bool {
        // Simple comparison against threshold
        isGreater : bool = input > threshold;
        return isGreater;
    }
    
    // Conditional evaluations
    category : char = evaluate(25);    // Returns 'M' (Medium)
    comparison : bool = compare(30);   // Returns True (30 > 20)
}

// FINAL STATE: value=25, threshold=20, category='M', comparison=True
