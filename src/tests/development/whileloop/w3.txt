class nestedLoop {
	x: int = 0;
	i: int = 0;
	j: int = 0;
	k: int = 0;
	l: int = 0;
	while (i < 5) {
		j = 0;
		while(j < 5) {
			k = 0;
			while (k < 5) {
				l=0;
				while (l < 2) {
					x = x + 1;
					l = l + 1;
				}
				k =k+ 1;
			}
			j = j + 1;
		}
		i = i + 1;
	}
	
}
