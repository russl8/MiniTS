// FEATURE DEMONSTRATED: Variable Assignment and Functions
// KEY CONCEPTS: Variable updates, function parameters, return values

class Student {
    studentId : int = 12345;
    grade : int = 85;
    credits : int = 12;
    
    function updateGrade(newGrade : int) : int {
        // Update student's grade
        grade = newGrade;
        return grade;
    }
    
    function addCredits(additionalCredits : int) : int {
        // Add credits to current total
        credits = credits + additionalCredits;
        return credits;
    }
    
    // Function calls to update student data
    finalGrade : int = updateGrade(92);     // Updates grade to 92
    totalCredits : int = addCredits(3);     // Adds 3 credits (12 + 3 = 15)
}

// FINAL STATE: studentId=12345, grade=92, credits=15, finalGrade=92, totalCredits=15
